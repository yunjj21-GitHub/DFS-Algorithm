// 백준 13023 : ABCDE (DFS탐색 이용)
#include <iostream>
#include <string>
#include <vector>

using namespace std;

int n, m;
vector<int> arr[2000];
bool chk[2000];
bool ans = false;

void DFS(int x, int depth) {
	if (ans) return;

	if (chk[x]) return;
	chk[x] = true;

	if (depth == 5) ans = true;

	for (int i = 0; i < arr[x].size(); i++) {
		int y = arr[x][i];
		DFS(y, depth + 1);
	}
	chk[x] = false; 
}

int main() {
	cin >> n >> m;
	for (int i = 0; i < m; i++) {
		int x, y;
		cin >> x >> y;
		arr[x].push_back(y);
		arr[y].push_back(x);
	}

	for (int i = 0; i < n; i++) { // 0-1, 2-3-4-5-6과 같은 경우 처리
		DFS(i, 1);
		if (ans) break;
	}

	cout << ans << '\n';
}
