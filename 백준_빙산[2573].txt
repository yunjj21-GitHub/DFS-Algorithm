// 백준 2573 : 빙산
#include <vector>
#include <string>
#include <iostream>
#include <algorithm>

using namespace std;

int N, M;
int map[300][300];
bool chk[300][300];
int dy[4] = { 0, 0, 1, -1 };
int dx[4] = { 1, -1, 0, 0 };

void Melt() {
	int oldMap[300][300];
	for (int i = 0; i < N; i++) {
		for (int j = 0; j < M; j++) {
			oldMap[i][j] = map[i][j];
		}
	}

	for (int y = 0; y < N; y++) {
		for (int x = 0; x < M; x++) {
			if (oldMap[y][x] == 0) continue;

			int cnt = 0;
			for (int i = 0; i < 4; i++) {
				int adjY = y + dy[i];
				int adjX = x + dx[i];

				if (oldMap[adjY][adjX] == 0) cnt++;
			}
			map[y][x] = max(oldMap[y][x] - cnt, 0);
		}
	}
}

void DFS(int y, int x) {
	if (chk[y][x]) return;

	chk[y][x] = true;
	for (int i = 0; i < 4; i++) {
		int nxtY = y + dy[i];
		int nxtX = x + dx[i];

		if (map[nxtY][nxtX] > 0) {
			DFS(nxtY, nxtX);
		}
	}
}

int main() {
	cin >> N >> M;
	for (int i = 0; i < N; i++) {
		for (int j = 0; j < M; j++) {
			cin >> map[i][j];
		}
	}
	
	int ans = 0;
	while (1) {
		int cnt = 0;
		for (int i = 0; i < N; i++) {
			for (int j = 0; j < M; j++) {
				if (map[i][j] > 0 && !chk[i][j]) {
					DFS(i, j);
					cnt++;
				}
			}
		}

		if (cnt > 1) { // 빙산이 한덩이 이상일 경우
			cout << ans << '\n';
			return 0;
		}
		else if (cnt == 0) { // 빙산이 모두 녹아버린 경우
			cout << 0 << '\n';
			return 0;
		}

		// chk배열 초기화
		for (int i = 0; i < N; i++) {
			for (int j = 0; j < M; j++) {
				chk[i][j] = false;
			}
		}

		Melt(); // 빙산을 녹인다.

		ans++;
	}
}
